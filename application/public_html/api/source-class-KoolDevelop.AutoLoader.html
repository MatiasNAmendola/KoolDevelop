 81:     private function underscore_path($path) {
 82:         return preg_replace_callback('/.[A-Z]/', function($matches) {
 83:             if ($matches[0][0] == '\\') {
 84:                 return strtolower($matches[0]);
 85:             } else {
 86:                 return $matches[0][0] . '_' . strtolower($matches[0][1]);
 87:             }    
 88:         }, $path);

 91:     /**
 92:      * PSR-0 style autoloader 
 93:      * 
 94:      * @see: https://github.com/php-fig/fig-standards/blob/master/accepted/PSR-0.md
 95:      * 
 96:      * @param string $classname Classname
 97:      * 
 98:      * @return boolean Class loaded
 99:      */
100:     private function autoloadPSR0($classname) {
101:         
102:         $classname = ltrim($classname, '\\');
103:         $filename  = '';
104:         $namespace = '';
105:         if ($last_ns_pos = strripos($classname, '\\')) {
106:             $namespace = substr($classname, 0, $last_ns_pos);
107:             $classname = substr($classname, $last_ns_pos + 1);
108:             $filename  = LIBS_PATH . DS . str_replace('\\', DS, $namespace) . DS;
109:         }        
110:         $filename .= str_replace('_', DS, $classname) . '.php';
111: 
112:         if (file_exists($filename)) {
113:             require $filename;
114:             return true;
115:         }
116:         
117:         return false;
118:     }

120:     /**
121:      * Callback for spl_autoload_register
122:      *
123:      * @param string $classname Class name to load
124:      *
125:      * @return bool Class loaded
126:      */
127:     protected function autoload($classname) {
128:         
129:         // First use PSR-0 autoloader
130:         foreach($this->Vendors as $vendor) {
131:             if (\strpos($classname, $vendor) === 0) {
132:                 if ($this->autoloadPSR0($classname)) {
133:                     return true;
134:                 }
135:             }
136:         }
137:         
138:         $classname = str_replace('\\\\', '\\', '\\' . \str_replace('_', '\\', $classname));
139: 
140:         // Loop trough Prefix Mappings
141:         foreach ($this->PrefixMappings as $prefix => $mappings) {
142:             if (\strpos($classname, $prefix) === 0) {
143:                 foreach ($mappings as $mapping) {                    
144:                     
145:                     // Underscore path elements and don't touch class                    
146:                     $_classpath = explode('\\', \substr($classname, strlen($prefix)));
147:                     $_classfile = array_pop($_classpath);                    
148:                     $_classpath = $this->underscore_path('\\' . join('\\', $_classpath)) . '\\' . $_classfile;
149:                     
150:                     $filename = $mapping . DS . \str_replace(array('\\'), DS, $_classpath) . '.php';                
151:                     
152:                     if (\file_exists($filename)) {
153:                         //echo " [FOUND] \n";
154:                         include $filename;
155:                         return true;
156:                     }
157:                     
158:                 }
159:             }
160:         }
161: 
162:         return false;
163:     }

181:      *
182:      * @param <type> $prefix    Prefix
183:      * @param <type> $directory Directory
184:      * @param bool   $prepend   True to prepend, false to append
185:      *
186:      * @return void
187:      */
188:     public function addMapping($prefix, $directory, $prepend = false) {
189: 
190:         if (!is_dir($directory)) {
191:             throw new \InvalidArgumentException(__f("Directory for AutoLoader Mapping does not exist.", 'kooldevelop'));
192:         }
193: 
194:         if (preg_match('/^[\\a-zA-Z0-9]+$/', $prefix) == 0) {
195:             throw new \InvalidArgumentException(__f("Invalid Prefix format for AutoLoader", 'kooldevelop'));
196:         }
197: 
198:         if (!isset($this->PrefixMappings[$prefix])) {
199:             $this->PrefixMappings[$prefix] = array();
200:         }
201: 
202:         if (\in_array($directory, $this->PrefixMappings[$prefix])) {
203:             return;
204:         }
205: 
206:         if ($prepend) {
207:             array_unshift($this->PrefixMappings[$prefix], $directory);
208:         } else {
209:             array_push($this->PrefixMappings[$prefix], $directory);
210:         }
211:     }
