 38:      * Find index of section in file contents, create

 42:      * @param string[] $file_contents File contents, exploded on newline
 43:      *
 44:      * @return void
 45:      */
 46:     private function findSectionIndex($section, &$file_contents) {
 47: 
 48:         foreach($file_contents as $index => &$row) {
 49:             if (trim($row) == '[' . $section . ']') {
 50:                 return $index;
 51:             }
 52:         }

 59:     /**
 60:      * Check if option is already in section
 61:      *
 62:      * @param string   $option        Option name
 63:      * @param int      $section_index Section index from findSectionIndex
 64:      * @param string[] $file_contents File contents, exploded on newline
 65:      *
 66:      * @return boolean Option exists
 67:      */
 68:     private function hasOption($option, $section_index, &$file_contents) {
 69:         
 70:         $file_contents_size = count($file_contents);
 71: 
 72:         // If section is at file end, we know that there are no options
 73:         if (($section_index+1) >= $file_contents_size) {
 74:             return false;
 75:         }
 76: 
 77:         for($x =($section_index+1); $x < $file_contents_size; $x++) {
 78: 
 79: 
 80:             // Start of new section
 81:             if (substr($file_contents[$x], 0, 1) == '[') {
 82:                 return false;
 83:             }
 84: 
 85:             // Find property in the form of
 86:             // property = ...
 87:             if (\preg_match('/^' . $option . '(\s*)=/', $file_contents[$x])) {
 88:                 return true;
 89:             }
 90: 
 91:             // Find property in the form of
 92:             // property[] = ...
 93:             if (\preg_match('/^' . $option . '\[\](\s*)=/', $file_contents[$x])) {
 94:                 return true;
 95:             }
 96: 
 97:         }
 98: 
 99:         return false;
