 33:     private function _convertOutputPath($input, $output) {
 34:         
 35:         if ($output[0] != DS) {
 36:             $input_path = str_replace('/', DS, pathinfo($input, PATHINFO_DIRNAME));
 37:             $path = str_replace('/', DS, pathinfo($output, PATHINFO_DIRNAME));
 38:             $file = pathinfo($output, PATHINFO_BASENAME);
 39:             
 40:             if (null === ($path = realpath($input_path . DS . $path))) {
 41:                 return false;
 42:             }            
 43:             $output = $path . DS . $file;                        
 44:         }
 45:         
 46:         return $output;
 47:     }

 49:     /**
 50:      * Convert an image
 51:      * 
 52:      * Converts the input image to output image using an transformation
 53:      * callback. When the output image exists this is used without reapplying
 54:      * the transformation.
 55:      * 
 56:      * The transformation callback recieves the input filename, an absolute
 57:      * path to the output filename and a reference to an \KoolDevelop\View\Image object
 58:      * 
 59:      * Output filename can be relative to the input filename, an absolute path
 60:      * is indicated by starting with the DIRECTORY_SEPARATOR
 61:      * 
 62:      * @param string   $input     Input image filename
 63:      * @param string   $output    Output image filename
 64:      * @param callable $transform Transformation callback
 65:      * 
 66:      * @return string Absolute path to image, false on failure
 67:      */
 68:     public function convert($input, $output, $transform) {
 69:         
 70:         if (false == ($output = $this->_convertOutputPath($input, $output))) {
 71:             return false;
 72:         }
 73:         
 74:         if (file_exists($output)) {
 75:             return $output;
 76:         }
 77:         
 78:         try {            
 79:             $image = new \KoolDevelop\View\Image();
 80:             call_user_func_array($transform, array($input, $output, &$image));
 81:         } catch(\Exception $e) {
 82:             return false;
 83:         }
 84:         
 85:         return $output;
 86:         
 87:     }
