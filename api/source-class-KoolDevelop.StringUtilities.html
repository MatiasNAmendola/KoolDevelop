111:         $underscored = "";

113: 
114:         // Transform
115:         for ($i = 0; $i < $length; $i++) {
116:             if ($i == 0) {
117:                 $underscored .= strtolower($camelcased[$i]);
118:             } else if ($camelcased[$i] == '_') {
